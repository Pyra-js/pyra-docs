# Config Types

<p className="text-xl text-gray-400 leading-relaxed">
  Complete TypeScript type definitions for Pyra.js configuration.
</p>

## UserConfig

The main configuration interface:

<GradientCode language="typescript">
{`interface UserConfig {
  /**
   * Project root directory
   * @default process.cwd()
   */
  root?: string;

  /**
   * Base public path
   * @default '/'
   */
  base?: string;

  /**
   * Application mode
   * @default 'development'
   */
  mode?: 'development' | 'production';

  /**
   * Define global constants
   */
  define?: Record<string, any>;

  /**
   * Array of plugins
   */
  plugins?: (Plugin | Plugin[])[];

  /**
   * Path resolution options
   */
  resolve?: ResolveOptions;

  /**
   * CSS options
   */
  css?: CSSOptions;

  /**
   * Development server options
   */
  server?: ServerOptions;

  /**
   * Build options
   */
  build?: BuildOptions;

  /**
   * Preview server options
   */
  preview?: PreviewOptions;

  /**
   * Dependency optimization options
   */
  optimizeDeps?: OptimizeDepsOptions;

  /**
   * SSR options
   */
  ssr?: SSROptions;
}`}
</GradientCode>

## ServerOptions

Development server configuration:

<GradientCode language="typescript">
{`interface ServerOptions {
  /**
   * Server port
   * @default 3000
   */
  port?: number;

  /**
   * Strict port - exit if port is in use
   * @default false
   */
  strictPort?: boolean;

  /**
   * Server host
   * @default false
   */
  host?: string | boolean;

  /**
   * Enable HTTPS
   */
  https?: boolean | {
    key: string;
    cert: string;
  };

  /**
   * Open browser on start
   * @default false
   */
  open?: boolean | string;

  /**
   * Proxy configuration
   */
  proxy?: Record<string, string | ProxyOptions>;

  /**
   * Enable CORS
   * @default true
   */
  cors?: boolean | CorsOptions;

  /**
   * HMR options
   */
  hmr?: boolean | {
    protocol?: string;
    host?: string;
    port?: number;
    path?: string;
    timeout?: number;
    overlay?: boolean;
  };

  /**
   * Custom middleware
   */
  middlewares?: Middleware[];
}`}
</GradientCode>

## BuildOptions

Production build configuration:

<GradientCode language="typescript">
{`interface BuildOptions {
  /**
   * Output directory
   * @default 'dist'
   */
  outDir?: string;

  /**
   * Assets directory (relative to outDir)
   * @default 'assets'
   */
  assetsDir?: string;

  /**
   * Static asset inline threshold (bytes)
   * @default 4096
   */
  assetsInlineLimit?: number;

  /**
   * Enable CSS code splitting
   * @default true
   */
  cssCodeSplit?: boolean;

  /**
   * Generate sourcemaps
   * @default false
   */
  sourcemap?: boolean | 'inline' | 'hidden';

  /**
   * Rollup options
   */
  rollupOptions?: RollupOptions;

  /**
   * Minification
   * @default 'esbuild'
   */
  minify?: boolean | 'esbuild' | 'terser';

  /**
   * Terser options (if minify is 'terser')
   */
  terserOptions?: TerserOptions;

  /**
   * Emit manifest.json
   * @default false
   */
  manifest?: boolean;

  /**
   * SSR build
   * @default false
   */
  ssr?: boolean | string;

  /**
   * Browser compatibility target
   * @default 'es2020'
   */
  target?: string | string[];

  /**
   * Chunk size warning limit (KB)
   * @default 500
   */
  chunkSizeWarningLimit?: number;

  /**
   * Watch mode
   */
  watch?: WatcherOptions | null;
}`}
</GradientCode>

## ResolveOptions

Module resolution configuration:

<GradientCode language="typescript">
{`interface ResolveOptions {
  /**
   * Path aliases
   */
  alias?: Record<string, string> | Array<{
    find: string | RegExp;
    replacement: string;
  }>;

  /**
   * Extensions to try when resolving imports
   * @default ['.mjs', '.js', '.ts', '.jsx', '.tsx', '.json']
   */
  extensions?: string[];

  /**
   * Conditions for package.json exports
   */
  conditions?: string[];

  /**
   * Main fields in package.json
   * @default ['module', 'jsnext:main', 'jsnext']
   */
  mainFields?: string[];

  /**
   * Preserve symlinks
   * @default false
   */
  preserveSymlinks?: boolean;
}`}
</GradientCode>

## CSSOptions

CSS processing configuration:

<GradientCode language="typescript">
{`interface CSSOptions {
  /**
   * CSS modules options
   */
  modules?: {
    scopeBehaviour?: 'global' | 'local';
    generateScopedName?: string | ((name: string, filename: string) => string);
    hashPrefix?: string;
    localsConvention?: 'camelCase' | 'camelCaseOnly' | 'dashes' | 'dashesOnly';
  };

  /**
   * PostCSS options
   */
  postcss?: string | {
    plugins?: any[];
    [key: string]: any;
  };

  /**
   * CSS preprocessor options
   */
  preprocessorOptions?: {
    scss?: any;
    sass?: any;
    less?: any;
    stylus?: any;
  };
}`}
</GradientCode>

## ProxyOptions

Proxy configuration for dev server:

<GradientCode language="typescript">
{`interface ProxyOptions {
  /**
   * Target URL
   */
  target: string;

  /**
   * Change origin header
   * @default false
   */
  changeOrigin?: boolean;

  /**
   * Rewrite path
   */
  rewrite?: (path: string) => string;

  /**
   * Configure proxy
   */
  configure?: (proxy: any, options: ProxyOptions) => void;

  /**
   * WebSocket proxy
   * @default false
   */
  ws?: boolean;

  /**
   * Bypass proxy for certain requests
   */
  bypass?: (req: any, res: any, options: ProxyOptions) => void | null | undefined | false | string;
}`}
</GradientCode>

## SSROptions

Server-side rendering configuration:

<GradientCode language="typescript">
{`interface SSROptions {
  /**
   * External modules for SSR
   */
  external?: string[];

  /**
   * Modules to not externalize for SSR
   */
  noExternal?: string | RegExp | (string | RegExp)[];

  /**
   * SSR target
   * @default 'node'
   */
  target?: 'node' | 'webworker';
}`}
</GradientCode>

## Usage Example

<GradientCode language="typescript" title="pyra.config.ts">
{`import { defineConfig, UserConfig } from 'pyra';
import react from '@pyra/plugin-react';

export default defineConfig({
  plugins: [react()],
  
  resolve: {
    alias: {
      '@': '/src',
      '@components': '/src/components'
    }
  },
  
  server: {
    port: 3000,
    host: true,
    proxy: {
      '/api': {
        target: 'http://localhost:4000',
        changeOrigin: true
      }
    }
  },
  
  build: {
    outDir: 'dist',
    sourcemap: true,
    rollupOptions: {
      output: {
        manualChunks: {
          vendor: ['react', 'react-dom']
        }
      }
    }
  }
} satisfies UserConfig);`}
</GradientCode>

<Callout type="tip">
  Use `satisfies UserConfig` for type checking while preserving literal types.
</Callout>
