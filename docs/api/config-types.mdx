# Configuration Types

Complete TypeScript type definitions for `pyra.config.ts`.

## PyraConfig

The main configuration object for Pyra.

<GradientCode language="typescript">
interface PyraConfig {
  /** Project root directory */
  root?: string;
  
  /** Base public path */
  base?: string;
  
  /** Build-specific options */
  build?: BuildOptions;
  
  /** Dev server options */
  server?: ServerOptions;
  
  /** Preview server options */
  preview?: PreviewOptions;
  
  /** Dependency optimization options */
  optimizeDeps?: DepOptimizationOptions;
  
  /** Module resolution options */
  resolve?: ResolveOptions;
  
  /** CSS options */
  css?: CSSOptions;
  
  /** Plugin array */
  plugins?: PyraPlugin[];
  
  /** Define global constants */
  define?: Record<string, any>;
  
  /** Environment variables prefix */
  envPrefix?: string | string[];
  
  /** Public directory */
  publicDir?: string;
  
  /** Cache directory */
  cacheDir?: string;
}
</GradientCode>

## BuildOptions

Options for production builds.

<GradientCode language="typescript">
interface BuildOptions {
  /** Output directory */
  outDir?: string;
  
  /** Assets directory (relative to outDir) */
  assetsDir?: string;
  
  /** Inline assets smaller than this (bytes) */
  assetsInlineLimit?: number;
  
  /** Generate sourcemaps */
  sourcemap?: boolean | 'inline' | 'hidden';
  
  /** Rollup options */
  rollupOptions?: RollupOptions;
  
  /** Minify output */
  minify?: boolean | 'terser' | 'esbuild';
  
  /** Target browsers */
  target?: string | string[];
  
  /** Emit manifest.json */
  manifest?: boolean | string;
  
  /** Enable/disable CSS code splitting */
  cssCodeSplit?: boolean;
  
  /** Chunk size warning limit (kB) */
  chunkSizeWarningLimit?: number;
  
  /** Watch mode */
  watch?: WatcherOptions | null;
}
</GradientCode>

<Callout type="note">
Most build options have sensible defaults. You typically only need to configure `outDir` and `sourcemap`.
</Callout>

## ServerOptions

Development server configuration.

<GradientCode language="typescript">
interface ServerOptions {
  /** Server host */
  host?: string | boolean;
  
  /** Server port */
  port?: number;
  
  /** Strict port (exit if port is in use) */
  strictPort?: boolean;
  
  /** Enable CORS */
  cors?: boolean | CorsOptions;
  
  /** Open browser on start */
  open?: boolean | string;
  
  /** Proxy configuration */
  proxy?: Record<string, string | ProxyOptions>;
  
  /** HMR options */
  hmr?: boolean | HmrOptions;
  
  /** Middleware mode */
  middlewareMode?: boolean | 'ssr' | 'html';
  
  /** File system options */
  fs?: FileSystemOptions;
  
  /** Origin for asset URLs */
  origin?: string;
}
</GradientCode>

## ResolveOptions

Module resolution configuration.

<GradientCode language="typescript">
interface ResolveOptions {
  /** Path aliases */
  alias?: Record<string, string> | Array<{ find: string | RegExp; replacement: string }>;
  
  /** Extensions to try */
  extensions?: string[];
  
  /** Conditions for package.json exports */
  conditions?: string[];
  
  /** Main fields in package.json */
  mainFields?: string[];
  
  /** Preserve symlinks */
  preserveSymlinks?: boolean;
}
</GradientCode>

### Example Usage

<GradientCode language="typescript">
import { defineConfig } from 'pyra';
import path from 'path';

export default defineConfig({
  resolve: {
    alias: {
      '@': path.resolve(__dirname, './src'),
      '@components': path.resolve(__dirname, './src/components'),
      '@utils': path.resolve(__dirname, './src/utils'),
    },
    extensions: ['.ts', '.tsx', '.js', '.jsx', '.json'],
  },
});
</GradientCode>

## DepOptimizationOptions

Dependency pre-bundling configuration.

<GradientCode language="typescript">
interface DepOptimizationOptions {
  /** Dependencies to include */
  include?: string[];
  
  /** Dependencies to exclude */
  exclude?: string[];
  
  /** Esbuild options */
  esbuildOptions?: EsbuildOptions;
  
  /** Force optimization */
  force?: boolean;
  
  /** Disabled optimization */
  disabled?: boolean | 'build' | 'dev';
}
</GradientCode>

### Example Usage

<GradientCode language="typescript">
export default defineConfig({
  optimizeDeps: {
    include: ['react', 'react-dom', 'lodash-es'],
    exclude: ['@my-local-package'],
    esbuildOptions: {
      target: 'es2020',
    },
  },
});
</GradientCode>

<Callout type="tip">
Add dependencies to `include` if they're slow to load or cause issues during development.
</Callout>

## PyraPlugin

Plugin interface for extending Pyra.

<GradientCode language="typescript">
interface PyraPlugin {
  /** Plugin name */
  name: string;
  
  /** Enforce plugin order */
  enforce?: 'pre' | 'post';
  
  /** Apply plugin conditionally */
  apply?: 'serve' | 'build' | ((config: PyraConfig, env: ConfigEnv) => boolean);
  
  /** Modify config */
  config?: (config: PyraConfig, env: ConfigEnv) => PyraConfig | null | void | Promise<PyraConfig | null | void>;
  
  /** Resolved config hook */
  configResolved?: (config: ResolvedConfig) => void | Promise<void>;
  
  /** Configure server */
  configureServer?: (server: PyraDevServer) => (() => void) | void | Promise<(() => void) | void>;
  
  /** Transform index.html */
  transformIndexHtml?: (html: string, ctx: IndexHtmlTransformContext) => string | HtmlTagDescriptor[] | Promise<string | HtmlTagDescriptor[]>;
  
  /** Resolve module ID */
  resolveId?: (source: string, importer: string | undefined, options: ResolveIdOptions) => string | null | Promise<string | null>;
  
  /** Load module */
  load?: (id: string) => string | null | Promise<string | null>;
  
  /** Transform module */
  transform?: (code: string, id: string) => TransformResult | Promise<TransformResult>;
  
  /** Handle HMR update */
  handleHotUpdate?: (ctx: HmrContext) => Array<ModuleNode> | void | Promise<Array<ModuleNode> | void>;
  
  /** Build start hook */
  buildStart?: (options: NormalizedInputOptions) => void | Promise<void>;
  
  /** Build end hook */
  buildEnd?: (error?: Error) => void | Promise<void>;
  
  /** Generate bundle hook */
  generateBundle?: (options: NormalizedOutputOptions, bundle: OutputBundle) => void | Promise<void>;
}
</GradientCode>

<Callout type="note">
Pyra plugins are compatible with Rollup plugins. Most Rollup plugins work in Pyra without modification.
</Callout>

## Usage Example

Complete configuration example:

<GradientCode language="typescript">
import { defineConfig } from 'pyra';
import react from '@pyra/plugin-react';
import path from 'path';

export default defineConfig({
  root: process.cwd(),
  base: '/',
  publicDir: 'public',
  cacheDir: 'node_modules/.pyra',
  
  resolve: {
    alias: {
      '@': path.resolve(__dirname, './src'),
    },
    extensions: ['.ts', '.tsx', '.js', '.jsx'],
  },
  
  server: {
    port: 5173,
    host: true,
    open: true,
    cors: true,
    hmr: {
      overlay: true,
    },
    proxy: {
      '/api': {
        target: 'http://localhost:8080',
        changeOrigin: true,
      },
    },
  },
  
  build: {
    outDir: 'dist',
    sourcemap: true,
    minify: 'esbuild',
    target: 'es2020',
    rollupOptions: {
      output: {
        manualChunks: {
          vendor: ['react', 'react-dom'],
        },
      },
    },
  },
  
  optimizeDeps: {
    include: ['react', 'react-dom'],
  },
  
  plugins: [react()],
  
  define: {
    __APP_VERSION__: JSON.stringify('1.0.0'),
  },
  
  envPrefix: 'PYRA_',
});
</GradientCode>
